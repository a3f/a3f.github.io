333 * compute 2t syndromes of ecc polynomial, i.e. ecc(a^j) for j=1..2t
334 */
335static void compute_syndromes(struct bch_control *bch, uint32_t *ecc,
336 unsigned int *syn)
337{
338 int i, j, s;
339 unsigned int m;
340 uint32_t poly;
341 const int t = GF_T(bch)((bch)->t);
342
343 s = bch->ecc_bits;
344
345 /* make sure extra bits in last ecc word are cleared */
346 m = ((unsigned int)s) & 31;
347 if (m)
348 ecc[s/32] &= ~((1u << (32-m))-1);
349 memset(syn, 0, 2*t*sizeof(*syn));
350
351 /* compute v(a^j) for j=1 .. 2t-1 */
352 do {
353 poly = *ecc++;
354 s -= 32;
355 while (poly) {
356 i = deg(poly);
357 for (j = 0; j < 2*t; j += 2)
358 syn[j] ^= a_pow(bch, (j+1)*(i+s));
359
360 poly ^= (1 << i);
361 }
362 } while (s > 0);
363
364 /* v(a^(2j)) = v(a^j)^2 */
365 for (j = 0; j < t; j++)
366 syn[2*j+1] = gf_sqr(bch, syn[j]);
367}

430static int solve_linear_system(struct bch_control *bch, unsigned int *rows,
431 unsigned int *sol, int nsol)
432{

444 /* find suitable row for elimination */

475 return 0;
476
477 rows[r] = (p && (r == param[p-1])) ?
478 p--, 1u << (m-r) : rows[r-p];
479 }
480 }
481
482 if (nsol != (1 << k))
483 /* unexpected number of solutions */
484 return 0;
485
486 for (p = 0; p < nsol; p++) {
487 /* set parameters for p-th solution */
488 for (c = 0; c < k; c++)
489 rows[param[c]] = (rows[param[c]] & ~1)|((p >> c) & 1);
490
491 /* compute unique solution */
492 tmp = 0;
493 for (r = m-1; r >= 0; r--) {
494 mask = rows[r] & (tmp|1);
495 tmp |= parity(mask) << (m-r);
496 }
497 sol[p] = tmp >> 1;
498 }
499 return nsol;
500}

502/*
503 * this function builds and solves a linear system for finding roots of a degree
504 * 4 affine monic polynomial X^4+aX^2+bX+c over GF(2^m).
505 */
506static int find_affine4_roots(struct bch_control *bch, unsigned int a,
507 unsigned int b, unsigned int c,
508 unsigned int *roots)
509{
510 int i, j, k;
511 const int m = GF_M(bch)((bch)->m);
512 unsigned int mask = 0xff, t, rows[16] = {0,};
513
514 j = a_log(bch, b);
515 k = a_log(bch, a);
516 rows[0] = c;
517
518 /* buid linear system to solve X^4+aX^2+bX+c = 0 */
519 for (i = 0; i < m; i++) {
520 rows[i+1] = bch->a_pow_tab[4*i]^
521 (a ? bch->a_pow_tab[mod_s(bch, k)] : 0)^
522 (b ? bch->a_pow_tab[mod_s(bch, j)] : 0);
523 j++;
524 k += 2;
525 }
526 /*
527 * transpose 16x16 matrix before passing it to linear solver
528 * warning: this code assumes m < 16
529 */
530 for (j = 8; j != 0; j >>= 1, mask ^= (mask << j)) {
531 for (k = 0; k < 16; k = (k+j+1) & ~j) {
532 t = ((rows[k] >> j)^rows[k+j]) & mask;
533 rows[k] ^= (t << j);
534 rows[k+j] ^= t;
535 }
536 }
537 return solve_linear_system(bch, rows, roots, 4);
538}

555/*
556 * compute roots of a degree 2 polynomial over GF(2^m)
557 */
558static int find_poly_deg2_roots(struct bch_control *bch, struct gf_poly *poly,
559 unsigned int *roots)
560{
561 int n = 0, i, l0, l1, l2;
562 unsigned int u, v, r;
563
564 if (poly->c[0] && poly->c[1]) {
565
566 l0 = bch->a_log_tab[poly->c[0]];
567 l1 = bch->a_log_tab[poly->c[1]];
568 l2 = bch->a_log_tab[poly->c[2]];
569
570 /* using z=a/bX, transform aX^2+bX+c into z^2+z+u (u=ac/b^2) */
571 u = a_pow(bch, l0+l2+2*(GF_N(bch)((bch)->n)-l1));
572 /*
573 * let u = sum(li.a^i) i=0..m-1; then compute r = sum(li.xi):
574 * r^2+r = sum(li.(xi^2+xi)) = sum(li.(a^i+Tr(a^i).a^k)) =
575 * u + sum(li.Tr(a^i).a^k) = u+a^k.Tr(sum(li.a^i)) = u+a^k.Tr(u)
576 * i.e. r and r+1 are roots iff Tr(u)=0
577 */
578 r = 0;
579 v = u;
580 while (v) {
581 i = deg(v);
582 r ^= bch->xi_tab[i];
583 v ^= (1 << i);
584 }
585 /* verify root */
586 if ((gf_sqr(bch, r)^r) == u) {
587 /* reverse z=a/bX transformation and compute log(1/r) */
588 roots[n++] = modulo(bch, 2*GF_N(bch)((bch)->n)-l1-
589 bch->a_log_tab[r]+l2);
590 roots[n++] = modulo(bch, 2*GF_N(bch)((bch)->n)-l1-
591 bch->a_log_tab[r^1]+l2);
592 }
593 }
594 return n;
595}

597/*
598 * compute roots of a degree 3 polynomial over GF(2^m)
599 */
600static int find_poly_deg3_roots(struct bch_control *bch, struct gf_poly *poly,
601 unsigned int *roots)
602{
603 int i, n = 0;
604 unsigned int a, b, c, a2, b2, c2, e3, tmp[4];
605
606 if (poly->c[0]) {
607 /* transform polynomial into monic X^3 + a2X^2 + b2X + c2 */
608 e3 = poly->c[3];
609 c2 = gf_div(bch, poly->c[0], e3);
610 b2 = gf_div(bch, poly->c[1], e3);
611 a2 = gf_div(bch, poly->c[2], e3);
612
613 /* (X+a2)(X^3+a2X^2+b2X+c2) = X^4+aX^2+bX+c (affine) */
614 c = gf_mul(bch, a2, c2); /* c = a2c2 */
615 b = gf_mul(bch, a2, b2)^c2; /* b = a2b2 + c2 */
616 a = gf_sqr(bch, a2)^b2; /* a = a2^2 + b2 */
617
618 /* find the 4 roots of this affine polynomial */
619 if (find_affine4_roots(bch, a, b, c, tmp) == 4) {
620 /* remove a2 from final list of roots */
621 for (i = 0; i < 4; i++) {
622 if (tmp[i] != a2)
623 roots[n++] = a_ilog(bch, tmp[i]);
624 }
625 }
626 }
627 return n;
628}

630/*
631 * compute roots of a degree 4 polynomial over GF(2^m)
632 */
633static int find_poly_deg4_roots(struct bch_control *bch, struct gf_poly *poly,
634 unsigned int *roots)
635{
636 int i, l, n = 0;
637 unsigned int a, b, c, d, e = 0, f, a2, b2, c2, e4;
638
639 if (poly->c[0] == 0)
640 return 0;
641
642 /* transform polynomial into monic X^4 + aX^3 + bX^2 + cX + d */
643 e4 = poly->c[4];
644 d = gf_div(bch, poly->c[0], e4);
645 c = gf_div(bch, poly->c[1], e4);
646 b = gf_div(bch, poly->c[2], e4);
647 a = gf_div(bch, poly->c[3], e4);
648
649 /* use Y=1/X transformation to get an affine polynomial */
650 if (a) {
651 /* first, eliminate cX by using z=X+e with ae^2+c=0 */
652 if (c) {
653 /* compute e such that e^2 = c/a */
654 f = gf_div(bch, c, a);
655 l = a_log(bch, f);
656 l += (l & 1) ? GF_N(bch)((bch)->n) : 0;
657 e = a_pow(bch, l/2);
658 /*
659 * use transformation z=X+e:
660 * z^4+e^4 + a(z^3+ez^2+e^2z+e^3) + b(z^2+e^2) +cz+ce+d
661 * z^4 + az^3 + (ae+b)z^2 + (ae^2+c)z+e^4+be^2+ae^3+ce+d
662 * z^4 + az^3 + (ae+b)z^2 + e^4+be^2+d
663 * z^4 + az^3 + b'z^2 + d'
664 */
665 d = a_pow(bch, 2*l)^gf_mul(bch, b, f)^d;
666 b = gf_mul(bch, a, e)^b;
667 }
668 /* now, use Y=1/X to get Y^4 + b/dY^2 + a/dY + 1/d */
669 if (d == 0)
670 /* assume all roots have multiplicity 1 */
671 return 0;
672
673 c2 = gf_inv(bch, d);
674 b2 = gf_div(bch, a, d);
675 a2 = gf_div(bch, b, d);
676 } else {
677 /* polynomial is already affine */
678 c2 = d;
679 b2 = c;
680 a2 = b;
681 }
682 /* find the 4 roots of this affine polynomial */
683 if (find_affine4_roots(bch, a2, b2, c2, roots) == 4) {
684 for (i = 0; i < 4; i++) {
685 /* post-process roots (reverse transformations) */
686 f = a ? gf_inv(bch, roots[i]) : roots[i];
687 roots[i] = a_ilog(bch, f^e);
688 }
689 n = 4;
690 }
691 return n;
692}

707/*
708 * compute polynomial Euclidean division remainder in GF(2^m)[X]
709 */
710static void gf_poly_mod(struct bch_control *bch, struct gf_poly *a,
711 const struct gf_poly *b, int *rep)
712{
713 int la, p, m;
714 unsigned int i, j, *c = a->c;
715 const unsigned int d = b->deg;
716
717 if (a->deg < d)
718 return;
719
720 /* reuse or compute log representation of denominator */
721 if (!rep) {
722 rep = bch->cache;
723 gf_poly_logrep(bch, b, rep);
724 }
725
726 for (j = a->deg; j >= d; j--) {
727 if (c[j]) {
728 la = a_log(bch, c[j]);
729 p = j-d;
730 for (i = 0; i < d; i++, p++) {
731 m = rep[i];
732 if (m >= 0)
733 c[p] ^= bch->a_pow_tab[mod_s(bch,
734 m+la)];
735 }
736 }
737 }
738 a->deg = d-1;
739 while (!c[a->deg] && a->deg)
740 a->deg--;
741}

833/*
834 * factor a polynomial using Berlekamp Trace algorithm (BTA)
835 */
836static void factor_polynomial(struct bch_control *bch, int k, struct gf_poly *f,
837 struct gf_poly **g, struct gf_poly **h)
838{
839 struct gf_poly *f2 = bch->poly_2t[0];
840 struct gf_poly *q = bch->poly_2t[1];
841 struct gf_poly *tk = bch->poly_2t[2];
842 struct gf_poly *z = bch->poly_2t[3];
843 struct gf_poly *gcd;
844
845 dbg("factoring %s...\n", gf_poly_str(f))do {} while (0);
846
847 *g = f;
848 *h = NULL((void *)0);
849
850 /* tk = Tr(a^k.X) mod f */
851 compute_trace_bk_mod(bch, k, f, z, tk);
852
853 if (tk->deg > 0) {
854 /* compute g = gcd(f, tk) (destructive operation) */
855 gf_poly_copy(f2, f);
856 gcd = gf_poly_gcd(bch, f2, tk);
857 if (gcd->deg < f->deg) {
858 /* compute h=f/gcd(f,tk); this will modify f and q */
859 gf_poly_div(bch, f, gcd, q);
860 /* store g and h in-place (clobbering f) */
861 *h = &((struct gf_poly_deg1 *)f)[gcd->deg].poly;
862 gf_poly_copy(*g, gcd);
863 gf_poly_copy(*h, q);
864 }
865 }
866}

868/*
869 * find roots of a polynomial, using BTZ algorithm; see the beginning of this
870 * file for details
871 */
872static int find_poly_roots(struct bch_control *bch, unsigned int k,
873 struct gf_poly *poly, unsigned int *roots)
874{
875 int cnt;
876 struct gf_poly *f1, *f2;
877
878 switch (poly->deg) {
879 /* handle low degree polynomials with ad hoc techniques */
880 case 1:
881 cnt = find_poly_deg1_roots(bch, poly, roots);
882 break;
883 case 2:
884 cnt = find_poly_deg2_roots(bch, poly, roots);
885 break;
886 case 3:
887 cnt = find_poly_deg3_roots(bch, poly, roots);
888 break;
889 case 4:
890 cnt = find_poly_deg4_roots(bch, poly, roots);
891 break;
892 default:
893 /* factor polynomial using Berlekamp Trace Algorithm (BTA) */
894 cnt = 0;
895 if (poly->deg && (k <= GF_M(bch)((bch)->m))) {
896 factor_polynomial(bch, k, poly, &f1, &f2);
897 if (f1)
898 cnt += find_poly_roots(bch, k+1, f1, roots);
899 if (f2)
900 cnt += find_poly_roots(bch, k+1, f2, roots+cnt);
901 }
902 break;
903 }
904 return cnt;
905}

1079 int i, j, b, d;
1080 uint32_t data, hi, lo, *tab;
1081 const int l = BCH_ECC_WORDS(bch)(((((bch)->m)*((bch)->t)) + (32) - 1) / (32));
1082 const int plen = DIV_ROUND_UP(bch->ecc_bits+1, 32)(((bch->ecc_bits+1) + (32) - 1) / (32));
1083 const int ecclen = DIV_ROUND_UP(bch->ecc_bits, 32)(((bch->ecc_bits) + (32) - 1) / (32));
1084
1085 memset(bch->mod8_tab, 0, 4*256*l*sizeof(*bch->mod8_tab));
1086
1087 for (i = 0; i < 256; i++) {
1088 /* p(X)=i is a small polynomial of weight <= 8 */
1089 for (b = 0; b < 4; b++) {
1090 /* we want to compute (p(X).X^(8*b+deg(g))) mod g(X) */
1091 tab = bch->mod8_tab + (b*256+i)*l;
1092 data = i << (8*b);
1093 while (data) {
1094 d = deg(data);
1095 /* subtract X^d.g(X) from p(X).X^(8*b+deg(g)) */
1096 data ^= g[0] >> (31-d);
1097 for (j = 0; j < ecclen; j++) {
1098 hi = (d < 31) ? g[j] << (d+1) : 0;
1099 lo = (j+1 < plen) ?
1100 g[j+1] >> (31-d) : 0;
1101 tab[j] ^= hi|lo;
1102 }
1103 }
1104 }
1105 }
1106}

1108/*
1109 * build a base for factoring degree 2 polynomials
1110 */
1111static int build_deg2_base(struct bch_control *bch)
1112{
1113 const int m = GF_M(bch)((bch)->m);
1114 int i, j, r;
1115 unsigned int sum, x, y, remaining, ak = 0, xi[m];
1116
1117 /* find k s.t. Tr(a^k) = 1 and 0 <= k < m */
1118 for (i = 0; i < m; i++) {
1119 for (j = 0, sum = 0; j < m; j++)
1120 sum ^= a_pow(bch, i*(1 << j));
1121
1122 if (sum) {
1123 ak = bch->a_pow_tab[i];
1124 break;
1125 }
1126 }
1127 /* find xi, i=0..m-1 such that xi^2+xi = a^i+Tr(a^i).a^k */
1128 remaining = m;
1129 memset(xi, 0, sizeof(xi));
1130
1131 for (x = 0; (x <= GF_N(bch)((bch)->n)) && remaining; x++) {
1132 y = gf_sqr(bch, x)^x;
1133 for (i = 0; i < 2; i++) {
1134 r = a_log(bch, y);
1135 if (y && (r < m) && !xi[r]) {
1136 bch->xi_tab[r] = x;
1137 xi[r] = 1;
1138 remaining--;
1139 dbg("x%d = %x\n", r, x)do {} while (0);
1140 break;
1141 }
1142 y ^= ak;
1143 }
1144 }
1145 /* should not happen but check anyway */
1146 return remaining ? -1 : 0;
1147}

1159/*
1160 * compute generator polynomial for given (m,t) parameters.
1161 */
1162static uint32_t *compute_generator_polynomial(struct bch_control *bch)
1163{
1164 const unsigned int m = GF_M(bch)((bch)->m);
1165 const unsigned int t = GF_T(bch)((bch)->t);
1166 int n, err = 0;
1167 unsigned int i, j, nbits, r, word, *roots;
1168 struct gf_poly *g;
1169 uint32_t *genpoly;
1170
1171 g = bch_alloc(GF_POLY_SZ(m*t)(sizeof(struct gf_poly)+((m*t)+1)*sizeof(unsigned int)), &err);
1172 roots = bch_alloc((bch->n+1)*sizeof(*roots), &err);
1173 genpoly = bch_alloc(DIV_ROUND_UP(m*t+1, 32)(((m*t+1) + (32) - 1) / (32))*sizeof(*genpoly), &err);
1174
1175 if (err) {
1176 kfree(genpoly);
1177 genpoly = NULL((void *)0);
1178 goto finish;
1179 }
1180
1181 /* enumerate all roots of g(X) */
1182 memset(roots , 0, (bch->n+1)*sizeof(*roots));
1183 for (i = 0; i < t; i++) {
1184 for (j = 0, r = 2*i+1; j < m; j++) {
1185 roots[r] = 1;
1186 r = mod_s(bch, 2*r);
1187 }
1188 }
1189 /* build generator polynomial g(X) */
1190 g->deg = 0;
1191 g->c[0] = 1;
1192 for (i = 0; i < GF_N(bch)((bch)->n); i++) {
1193 if (roots[i]) {
1194 /* multiply g(X) by (X+root) */
1195 r = bch->a_pow_tab[i];
1196 g->c[g->deg+1] = 1;
1197 for (j = g->deg; j > 0; j--)
1198 g->c[j] = gf_mul(bch, g->c[j], r)^g->c[j-1];
1199
1200 g->c[0] = gf_mul(bch, g->c[0], r);
1201 g->deg++;
1202 }
1203 }
1204 /* store left-justified binary representation of g(X) */
1205 n = g->deg+1;
1206 i = 0;
1207
1208 while (n > 0) {
1209 nbits = (n > 32) ? 32 : n;
1210 for (j = 0, word = 0; j < nbits; j++) {
1211 if (g->c[n-1-j])
1212 word |= 1u << (31-j);
1213 }
1214 genpoly[i++] = word;
1215 n -= nbits;
1216 }
1217 bch->ecc_bits = g->deg;
1218
1219finish:
1220 kfree(g);
1221 kfree(roots);
1222
1223 return genpoly;
1224}

1226/**
1227 * init_bch - initialize a BCH encoder/decoder
1228 * @m: Galois field order, should be in the range 5-15
1229 * @t: maximum error correction capability, in bits
1230 * @prim_poly: user-provided primitive polynomial (or 0 to use default)
1231 *
1232 * Returns:
1233 * a newly allocated BCH control structure if successful, NULL otherwise
1234 *
1235 * This initialization can take some time, as lookup tables are built for fast
1236 * encoding/decoding; make sure not to call this function from a time critical
1237 * path. Usually, init_bch() should be called on module/driver init and
1238 * free_bch() should be called to release memory on exit.
1239 *
1240 * You may provide your own primitive polynomial of degree @m in argument
1241 * @prim_poly, or let init_bch() use its default polynomial.
1242 *
1243 * Once init_bch() has successfully returned a pointer to a newly allocated
1244 * BCH control structure, ecc length in bytes is given by member @ecc_bytes of
1245 * the structure.
1246 */
1247struct bch_control *init_bch(int m, int t, unsigned int prim_poly)
1248{
1249 int err = 0;
1250 unsigned int i, words;
1251 uint32_t *genpoly;
1252 struct bch_control *bch = NULL((void *)0);
1253
1254 const int min_m = 5;
1255 const int max_m = 15;
1256
1257 /* default primitive polynomials */
1258 static const unsigned int prim_poly_tab[] = {
1259 0x25, 0x43, 0x83, 0x11d, 0x211, 0x409, 0x805, 0x1053, 0x201b,
1260 0x402b, 0x8003,
1261 };
1262
1263#if defined(CONFIG_BCH_CONST_PARAMS)
1264 if ((m != (CONFIG_BCH_CONST_M)) || (t != (CONFIG_BCH_CONST_T))) {
1265 printkbarebox_printf(KERN_ERR"" "bch encoder/decoder was configured to support "
1266 "parameters m=%d, t=%d only!\n",
1267 CONFIG_BCH_CONST_M, CONFIG_BCH_CONST_T);
1268 goto fail;
1269 }
1270#endif
1271 if ((m < min_m) || (m > max_m))
1272 /*
1273 * values of m greater than 15 are not currently supported;
1274 * supporting m > 15 would require changing table base type
1275 * (uint16_t) and a small patch in matrix transposition
1276 */
1277 goto fail;
1278
1279 /* sanity checks */
1280 if ((t < 1) || (m*t >= ((1 << m)-1)))
1281 /* invalid t value */
1282 goto fail;
1283
1284 /* select a primitive polynomial for generating GF(2^m) */
1285 if (prim_poly == 0)
1286 prim_poly = prim_poly_tab[m-min_m];
1287
1288 bch = kzalloc(sizeof(*bch), GFP_KERNEL0);
1289 if (bch == NULL((void *)0))
1290 goto fail;
1291
1292 bch->m = m;
1293 bch->t = t;
1294 bch->n = (1 << m)-1;
1295 words = DIV_ROUND_UP(m*t, 32)(((m*t) + (32) - 1) / (32));
1296 bch->ecc_bytes = DIV_ROUND_UP(m*t, 8)(((m*t) + (8) - 1) / (8));
1297 bch->a_pow_tab = bch_alloc((1+bch->n)*sizeof(*bch->a_pow_tab), &err);
1298 bch->a_log_tab = bch_alloc((1+bch->n)*sizeof(*bch->a_log_tab), &err);
1299 bch->mod8_tab = bch_alloc(words*1024*sizeof(*bch->mod8_tab), &err);
1300 bch->ecc_buf = bch_alloc(words*sizeof(*bch->ecc_buf), &err);
1301 bch->ecc_buf2 = bch_alloc(words*sizeof(*bch->ecc_buf2), &err);
1302 bch->xi_tab = bch_alloc(m*sizeof(*bch->xi_tab), &err);
1303 bch->syn = bch_alloc(2*t*sizeof(*bch->syn), &err);
1304 bch->cache = bch_alloc(2*t*sizeof(*bch->cache), &err);
1305 bch->elp = bch_alloc((t+1)*sizeof(struct gf_poly_deg1), &err);
1306
1307 for (i = 0; i < ARRAY_SIZE(bch->poly_2t)(sizeof(bch->poly_2t) / sizeof((bch->poly_2t)[0]) + (sizeof
(struct { int:(-!!(__builtin_types_compatible_p(typeof((bch->
poly_2t)), typeof(&(bch->poly_2t)[0])))); }))); i++)
1308 bch->poly_2t[i] = bch_alloc(GF_POLY_SZ(2*t)(sizeof(struct gf_poly)+((2*t)+1)*sizeof(unsigned int)), &err);
1309
1310 if (err)
1311 goto fail;
1312
1313 err = build_gf_tables(bch, prim_poly);
1314 if (err)
1315 goto fail;
1316
1317 /* use generator polynomial for computing encoding tables */
1318 genpoly = compute_generator_polynomial(bch);
1319 if (genpoly == NULL((void *)0))
1320 goto fail;
1321
1322 build_mod8_tables(bch, genpoly);
1323 kfree(genpoly);
1324
1325 err = build_deg2_base(bch);
1326 if (err)
1327 goto fail;
1328
1329 return bch;
1330
1331fail:
1332 free_bch(bch);
1333 return NULL((void *)0);
1334}
