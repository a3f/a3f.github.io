16 * Adds a character devices :
17 * - mtdraw<N>
18 *
19 * Device mtd_raw<N> provides acces to the MTD "pages+OOB". For example if a MTD
20 * has pages of 512 bytes and OOB of 16 bytes, mtd_oob<N> will be made of blocks
21 * of 528 bytes, with page data being followed by OOB.
22 * The layout will be: <page0> <oob0> <page1> <oob1> ... <pageN> <oobN>.
23 * This means that a read at offset 516 of 20 bytes will give the 12 last bytes
24 * of the OOB of page0, and the 8 first bytes of page1.
25 * Same thing applies for writes, which have to be page+oob aligned (ie. offset
26 * and size should be multiples of (mtd->writesize + mtd->oobsize)).
27 */

41/**
42 * mtdraw - mtdraw device private data
43 * @cdev: character device "mtdraw<N>"
44 * @mtd: MTD device to handle read/writes/erases
45 *
46 * @writebuf: buffer to handle unaligned writes (ie. writes of sizes which are
47 * not multiples of MTD (writesize+oobsize)
48 * @write_fill: number of bytes in writebuf
49 * @write_ofs: offset in character device (mtdraw) where last write(s) stored
50 * bytes because of unaligned writes (ie. remain of writesize+oobsize write)
51 * @rps: raw page size (writesize+oobsize)
52 *
53 * The mtdraw device must allow unaligned writes. This is enabled by a write buffer which gathers data to issue mtd->write_oob() with full page+oob data.
54 * Suppose writesize=512, oobsize=16.
55 * A first write of 512 bytes triggers:
56 * - write_ofs = offset of write()
57 * - write_fill = 512
58 * - copy of the 512 provided bytes into writebuf
59 * - no actual mtd->write if done
60 * A second write of 512 bytes triggers:
61 * - copy of the 16 first bytes into writebuf
62 * - a mtd_write_oob() from writebuf
63 * - empty writebuf
64 * - copy the remaining 496 bytes into writebuf
65 * => write_fill = 496, write_ofs = offset + 528
66 * Etc ...
67 */
68struct mtdraw {
69 struct cdev cdev;
70 struct mtd_info *mtd;
71 void *writebuf;
72 int write_fill;
73 int write_ofs;
74 unsigned int rps;
75};
