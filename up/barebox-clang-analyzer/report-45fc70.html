29/*
30 * DataFlash is a kind of SPI flash. Most AT45 chips have two buffers in
31 * each chip, which may be used for double buffered I/O; but this driver
32 * doesn't (yet) use these for any kind of i/o overlap or prefetching.
33 *
34 * Sometimes DataFlash is packaged in MMC-format cards, although the
35 * MMC stack can't (yet?) distinguish between MMC and DataFlash
36 * protocols during enumeration.
37 */

449static ssize_t otp_read(struct spi_device *spi, unsigned base,
450 u8 *buf, loff_t off, size_t len)
451{
452 struct spi_message m;
453 size_t l;
454 u8 *scratch;
455 struct spi_transfer t;
456 int status;
457
458 if (off > 64)
459 return -EINVAL22;
460
461 if ((off + len) > 64)
462 len = 64 - off;
463
464 spi_message_init(&m);
465
466 l = 4 + base + off + len;
467 scratch = kzalloc(l, GFP_KERNEL0);
468 if (!scratch)
469 return -ENOMEM12;
470
471 /* OUT: OP_READ_SECURITY, 3 don't-care bytes, zeroes
472 * IN: ignore 4 bytes, data bytes 0..N (max 127)
473 */
474 scratch[0] = OP_READ_SECURITY0x77;
475
476 memset(&t, 0, sizeof t);
477 t.tx_buf = scratch;
478 t.rx_buf = scratch;
479 t.len = l;
480 spi_message_add_tail(&t, &m);
481
482 dataflash_waitready(spi);
483
484 status = spi_sync(spi, &m);
485 if (status >= 0) {
486 memcpy(buf, scratch + 4 + base + off, len);
487 status = len;
488 }
489
490 kfree(scratch);
491 return status;
492}

524static int dataflash_write_user_otp(struct mtd_info *mtd,
525 loff_t from, size_t len, size_t *retlen, u_char *buf)
526{
527 struct spi_message m;
528 const size_t l = 4 + 64;
529 u8 *scratch;
530 struct spi_transfer t;
531 struct dataflash *priv = mtd->priv;
532 int status;
533
534 if (len > 64)
535 return -EINVAL22;
536
537 /* Strictly speaking, we *could* truncate the write ... but
538 * let's not do that for the only write that's ever possible.
539 */
540 if ((from + len) > 64)
541 return -EINVAL22;
542
543 /* OUT: OP_WRITE_SECURITY, 3 zeroes, 64 data-or-zero bytes
544 * IN: ignore all
545 */
546 scratch = kzalloc(l, GFP_KERNEL0);
547 if (!scratch)
548 return -ENOMEM12;
549 scratch[0] = OP_WRITE_SECURITY0x9B;
550 memcpy(scratch + 4 + from, buf, len);
551
552 spi_message_init(&m);
553
554 memset(&t, 0, sizeof t);
555 t.tx_buf = scratch;
556 t.len = l;
557 spi_message_add_tail(&t, &m);
558
559 /* Write the OTP bits, if they've not yet been written.
560 * This modifies SRAM buffer1.
561 */
562 dataflash_waitready(priv->spi);
563 status = spi_sync(priv->spi, &m);
564
565 kfree(scratch);
566
567 if (status >= 0) {
568 status = 0;
569 *retlen = len;
570 }
571 return status;
572}

760static struct flash_info * jedec_probe(struct spi_device *spi)
761{
762 int ret;
763 u8 code = OP_READ_ID0x9F;
764 u64 jedec;
765 u8 id[sizeof(jedec)] = {0};
766 const unsigned int id_size = 5;
767 struct flash_info *info;
768
769 /*
770 * JEDEC also defines an optional "extended device information"
771 * string for after vendor-specific data, after the three bytes
772 * we use here. Supporting some chips might require using it.
773 *
774 * If the vendor ID isn't Atmel's (0x1f), assume this call failed.
775 * That's not an error; only rev C and newer chips handle it, and
776 * only Atmel sells these chips.
777 */
778 ret = spi_write_then_read(spi, &code, 1, id, id_size);
779 if (ret < 0) {
780 dev_dbg(&spi->dev, "error %d reading JEDEC ID\n", ret)({ (7) <= 6 ? dev_printf((7), ((&spi->dev)), ("error %d reading JEDEC ID\n"
), ret) : 0; });
781 return ERR_PTR(ret);
782 }
783
784 if (id[0] != CFI_MFR_ATMEL0x1F)
785 return NULL((void *)0);
786
787 jedec = be64_to_cpup__be64_to_cpup((__be64 *)id);
788
789 /*
790 * First, try to match device using extended device
791 * information
792 */
793 info = jedec_lookup(spi, jedec >> DATAFLASH_SHIFT_EXTID24, true);
794 if (!IS_ERR(info))
795 return info;
796 /*
797 * If that fails, make another pass using regular ID
798 * information
799 */
800 info = jedec_lookup(spi, jedec >> DATAFLASH_SHIFT_ID40, false);
801 if (!IS_ERR(info))
802 return info;
803
804 /*
805 * Treat other chips as errors ... we won't know the right page
806 * size (it might be binary) even when we can tell which density
807 * class is involved (legacy chip id scheme).
808 */
809 dev_warn(&spi->dev, "JEDEC id %016llx not handled\n", jedec)({ (4) <= 6 ? dev_printf((4), ((&spi->dev)), ("JEDEC id %016llx not handled\n"
), jedec) : 0; });
810 return ERR_PTR(-ENODEV19);
811}

813/*
814 * Detect and initialize DataFlash device, using JEDEC IDs on newer chips
815 * or else the ID code embedded in the status bits:
816 *
817 * Device Density ID code #Pages PageSize Offset
818 * AT45DB011B 1Mbit (128K) xx0011xx (0x0c) 512 264 9
819 * AT45DB021B 2Mbit (256K) xx0101xx (0x14) 1024 264 9
820 * AT45DB041B 4Mbit (512K) xx0111xx (0x1c) 2048 264 9
821 * AT45DB081B 8Mbit (1M) xx1001xx (0x24) 4096 264 9
822 * AT45DB0161B 16Mbit (2M) xx1011xx (0x2c) 4096 528 10
823 * AT45DB0321B 32Mbit (4M) xx1101xx (0x34) 8192 528 10
824 * AT45DB0642 64Mbit (8M) xx111xxx (0x3c) 8192 1056 11
825 * AT45DB1282 128Mbit (16M) xx0100xx (0x10) 16384 1056 11
826 */
827static int dataflash_probe(struct device_d *dev)
828{
829 struct spi_device *spi = (struct spi_device *)dev->type_data;
830 int status;
831 struct flash_info *info;
832
833 /*
834 * Try to detect dataflash by JEDEC ID.
835 * If it succeeds we know we have either a C or D part.
836 * D will support power of 2 pagesize option.
837 * Both support the security register, though with different
838 * write procedures.
839 */
840 info = jedec_probe(spi);
841 if (IS_ERR(info))
842 return PTR_ERR(info);
843 if (info != NULL((void *)0))
844 return add_dataflash_otp(spi, info->name, info->nr_pages,
845 info->pagesize, info->pageoffset,
846 (info->flags & SUP_POW2PS0x0002) ? 'd' : 'c');
847
848 /*
849 * Older chips support only legacy commands, identifing
850 * capacity using bits in the status byte.
851 */
852 status = dataflash_status(spi);
853 if (status <= 0 || status == 0xff) {
854 dev_dbg(&spi->dev, "status error %d\n", status)({ (7) <= 6 ? dev_printf((7), ((&spi->dev)), ("status error %d\n"
), status) : 0; });
855 if (status == 0 || status == 0xff)
856 status = -ENODEV19;
857 return status;
858 }
859
860 /* if there's a device there, assume it's dataflash.
861 * board setup should have set spi->max_speed_max to
862 * match f(car) for continuous reads, mode 0 or 3.
863 */
864 switch (status & 0x3c) {
865 case 0x0c: /* 0 0 1 1 x x */
866 status = add_dataflash(spi, "AT45DB011B", 512, 264, 9);
867 break;
868 case 0x14: /* 0 1 0 1 x x */
869 status = add_dataflash(spi, "AT45DB021B", 1024, 264, 9);
870 break;
871 case 0x1c: /* 0 1 1 1 x x */
872 status = add_dataflash(spi, "AT45DB041x", 2048, 264, 9);
873 break;
874 case 0x24: /* 1 0 0 1 x x */
875 status = add_dataflash(spi, "AT45DB081B", 4096, 264, 9);
876 break;
877 case 0x2c: /* 1 0 1 1 x x */
878 status = add_dataflash(spi, "AT45DB161x", 4096, 528, 10);
879 break;
880 case 0x34: /* 1 1 0 1 x x */
881 status = add_dataflash(spi, "AT45DB321x", 8192, 528, 10);
882 break;
883 case 0x38: /* 1 1 1 x x x */
884 case 0x3c:
885 status = add_dataflash(spi, "AT45DB642x", 8192, 1056, 11);
886 break;
887 /* obsolete AT45DB1282 not (yet?) supported */
888 default:
889 dev_dbg(&spi->dev, "unsupported device (%x)\n", status & 0x3c)({ (7) <= 6 ? dev_printf((7), ((&spi->dev)), ("unsupported device (%x)\n"
), status & 0x3c) : 0; });
890 status = -ENODEV19;
891 }
892
893 if (status < 0)
894 dev_dbg(&spi->dev, "add_dataflash --> %d\n", status)({ (7) <= 6 ? dev_printf((7), ((&spi->dev)), ("add_dataflash --> %d\n"
), status) : 0; });
895
896 return status;
897}
