35/*
36 * Inventra (Multipoint) Dual-Role Controller Driver for Linux.
37 *
38 * This consists of a Host Controller Driver (HCD) and a peripheral
39 * controller driver implementing the "Gadget" API; OTG support is
40 * in the works. These are normal Linux-USB controller drivers which
41 * use IRQs and have no dedicated thread.
42 *
43 * This version of the driver has only been used with products from
44 * Texas Instruments. Those products integrate the Inventra logic
45 * with other DMA, IRQ, and bus modules, as well as other logic that
46 * needs to be reflected in this driver.
47 *
48 *
49 * NOTE: the original Mentor code here was pretty much a collection
50 * of mechanisms that don't seem to have been fully integrated/working
51 * for *any* Linux kernel version. This version aims at Linux 2.6.now,
52 * Key open issues include:
53 *
54 * - Lack of host-side transaction scheduling, for all transfer types.
55 * The hardware doesn't do it; instead, software must.
56 *
57 * This is not an issue for OTG devices that don't support external
58 * hubs, but for more "normal" USB hosts it's a user issue that the
59 * "multipoint" support doesn't scale in the expected ways. That
60 * includes DaVinci EVM in a common non-OTG mode.
61 *
62 * * Control and bulk use dedicated endpoints, and there's as
63 * yet no mechanism to either (a) reclaim the hardware when
64 * peripherals are NAKing, which gets complicated with bulk
65 * endpoints, or (b) use more than a single bulk endpoint in
66 * each direction.
67 *
68 * RESULT: one device may be perceived as blocking another one.
69 *
70 * * Interrupt and isochronous will dynamically allocate endpoint
71 * hardware, but (a) there's no record keeping for bandwidth;
72 * (b) in the common case that few endpoints are available, there
73 * is no mechanism to reuse endpoints to talk to multiple devices.
74 *
75 * RESULT: At one extreme, bandwidth can be overcommitted in
76 * some hardware configurations, no faults will be reported.
77 * At the other extreme, the bandwidth capabilities which do
78 * exist tend to be severely undercommitted. You can't yet hook
79 * up both a keyboard and a mouse to an external USB hub.
80 */

557/*
558 * configure a fifo; for non-shared endpoints, this may be called
559 * once for a tx fifo and once for an rx fifo.
560 *
561 * returns negative errno or offset for next fifo.
562 */
563static int
564fifo_setup(struct musb *musb, struct musb_hw_ep *hw_ep,
565 const struct musb_fifo_cfg *cfg, u16 offset)
566{
567 void __iomem *mbase = musb->mregs;
568 int size = 0;
569 u16 maxpacket = cfg->maxpacket;
570 u16 c_off = offset >> 3;
571 u8 c_size;
572
573 /* expect hw_ep has already been zero-initialized */
574
575 size = ffs(max(maxpacket, (u16) 8)({ typeof(maxpacket) _max1 = (maxpacket); typeof((u16) 8) _max2
= ((u16) 8); (void) (&_max1 == &_max2); _max1 > _max2
? _max1 : _max2; })) - 1;
576 maxpacket = 1 << size;
577
578 c_size = size - 3;
579 if (cfg->mode == BUF_DOUBLE) {
580 if ((offset + (maxpacket << 1)) >
581 (1 << (musb->config->ram_bits + 2)))
582 return -EMSGSIZE90;
583 c_size |= MUSB_FIFOSZ_DPB0x10;
584 } else {
585 if ((offset + maxpacket) > (1 << (musb->config->ram_bits + 2)))
586 return -EMSGSIZE90;
587 }
588
589 /* configure the FIFO */
590 musb_writeb(mbase, MUSB_INDEX0x0E, hw_ep->epnum);
591
592 /* EP0 reserved endpoint for control, bidirectional;
593 * EP1 reserved for bulk, two unidirectional halves.
594 */
595 if (hw_ep->epnum == 1)
596 musb->bulk_ep = hw_ep;
597 /* REVISIT error check: be sure ep0 can both rx and tx ... */
598 switch (cfg->style) {
599 case FIFO_TX:
600 musb_write_txfifosz(mbase, c_size);
601 musb_write_txfifoadd(mbase, c_off);
602 hw_ep->tx_double_buffered = !!(c_size & MUSB_FIFOSZ_DPB0x10);
603 hw_ep->max_packet_sz_tx = maxpacket;
604 break;
605 case FIFO_RX:
606 musb_write_rxfifosz(mbase, c_size);
607 musb_write_rxfifoadd(mbase, c_off);
608 hw_ep->rx_double_buffered = !!(c_size & MUSB_FIFOSZ_DPB0x10);
609 hw_ep->max_packet_sz_rx = maxpacket;
610 break;
611 case FIFO_RXTX:
612 musb_write_txfifosz(mbase, c_size);
613 musb_write_txfifoadd(mbase, c_off);
614 hw_ep->rx_double_buffered = !!(c_size & MUSB_FIFOSZ_DPB0x10);
615 hw_ep->max_packet_sz_rx = maxpacket;
616
617 musb_write_rxfifosz(mbase, c_size);
618 musb_write_rxfifoadd(mbase, c_off);
619 hw_ep->tx_double_buffered = hw_ep->rx_double_buffered;
620 hw_ep->max_packet_sz_tx = maxpacket;
621
622 hw_ep->is_shared_fifo = true;
623 break;
624 }
625
626 /* NOTE rx and tx endpoint irqs aren't managed separately,
627 * which happens to be ok
628 */
629 musb->epmask |= (1 << hw_ep->epnum);
630
631 return offset + (maxpacket << ((c_size & MUSB_FIFOSZ_DPB0x10) ? 1 : 0));
632}
