317static void *memalign_realloc(void *orig, size_t oldsize, size_t newsize)
318{
319 int align;
320 void *newbuf;
321
322 /*
323 * ARM Linux uses a single 1MiB section (with 1MiB alignment)
324 * for mapping the devicetree, so we are not allowed to cross
325 * 1MiB boundaries. This got fixed in the Kernel since v3.8-rc5
326 */
327 align = 1 << fls(newsize - 1);
328
329 if (!orig)
330 return memalign(align, newsize);
331
332 newbuf = memalign(align, newsize);
333 if (!newbuf) {
334 freebarebox_free(orig);
335 return NULL((void *)0);
336 }
337
338 memcpy(newbuf, orig, oldsize);
339 freebarebox_free(orig);
340 memset(newbuf + oldsize, 0, newsize - oldsize);
341
342 return newbuf;
343}
