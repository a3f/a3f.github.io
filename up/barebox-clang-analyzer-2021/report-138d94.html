102static int check_edid(unsigned char *edid)
103{
104 unsigned char *block = edid + ID_MANUFACTURER_NAME0x08, manufacturer[4];
105 unsigned char *b;
106 u32 model;
107 int i, fix = 0, ret = 0;
108
109 manufacturer[0] = ((block[0] & 0x7c) >> 2) + '@';
110 manufacturer[1] = ((block[0] & 0x03) << 3) +
111 ((block[1] & 0xe0) >> 5) + '@';
112 manufacturer[2] = (block[1] & 0x1f) + '@';
113 manufacturer[3] = 0;
114 model = block[2] + (block[3] << 8);
115
116 switch (fix) {
117 case FBMON_FIX_HEADER1:
118 for (i = 0; i < 8; i++) {
119 if (edid[i] != edid_v1_header[i]) {
120 ret = fix;
121 break;
122 }
123 }
124 break;
125 case FBMON_FIX_INPUT2:
126 b = edid + EDID_STRUCT_DISPLAY0x14;
127 /* Only if display is GTF capable will
128 the input type be reset to analog */
129 if (b[4] & 0x01 && b[0] & 0x80)
130 ret = fix;
131 break;
132 case FBMON_FIX_TIMINGS3:
133 b = edid + DETAILED_TIMING_DESCRIPTIONS_START0x36;
134 ret = fix;
135
136 for (i = 0; i < 4; i++) {
137 if (edid_is_limits_block(b)) {
138 ret = 0;
139 break;
140 }
141
142 b += DETAILED_TIMING_DESCRIPTION_SIZE18;
143 }
144
145 break;
146 }
147
148 if (ret)
149 printkbarebox_printf("fbmon: The EDID Block of "
150 "Manufacturer: %s Model: 0x%x is known to "
151 "be broken,\n", manufacturer, model);
152
153 return ret;
154}

156static void fix_edid(unsigned char *edid, int fix)
157{
158 int i;
159 unsigned char *b, csum = 0;
160
161 switch (fix) {
162 case FBMON_FIX_HEADER1:
163 printkbarebox_printf("fbmon: trying a header reconstruct\n");
164 memcpy(edid, edid_v1_header, 8);
165 break;
166 case FBMON_FIX_INPUT2:
167 printkbarebox_printf("fbmon: trying to fix input type\n");
168 b = edid + EDID_STRUCT_DISPLAY0x14;
169 b[0] &= ~0x80;
170 edid[127] += 0x80;
171 break;
172 case FBMON_FIX_TIMINGS3:
173 printkbarebox_printf("fbmon: trying to fix monitor timings\n");
174 b = edid + DETAILED_TIMING_DESCRIPTIONS_START0x36;
175 for (i = 0; i < 4; i++) {
176 if (!(edid_is_serial_block(b) ||
177 edid_is_ascii_block(b) ||
178 edid_is_monitor_block(b) ||
179 edid_is_timing_block(b))) {
180 b[0] = 0x00;
181 b[1] = 0x00;
182 b[2] = 0x00;
183 b[3] = 0xfd;
184 b[4] = 0x00;
185 b[5] = 60; /* vfmin */
186 b[6] = 60; /* vfmax */
187 b[7] = 30; /* hfmin */
188 b[8] = 75; /* hfmax */
189 b[9] = 17; /* pixclock - 170 MHz*/
190 b[10] = 0; /* GTF */
191 break;
192 }
193
194 b += DETAILED_TIMING_DESCRIPTION_SIZE18;
195 }
196
197 for (i = 0; i < EDID_LENGTH0x80 - 1; i++)
198 csum += edid[i];
199
200 edid[127] = 256 - csum;
201 break;
202 }
203}

205static int edid_checksum(unsigned char *edid)
206{
207 unsigned char csum = 0, all_null = 0;
208 int i, err = 0, fix = check_edid(edid);
209
210 if (fix)
211 fix_edid(edid, fix);
212
213 for (i = 0; i < EDID_LENGTH0x80; i++) {
214 csum += edid[i];
215 all_null |= edid[i];
216 }
217
218 if (csum == 0x00 && all_null) {
219 /* checksum passed, everything's good */
220 err = 1;
221 }
222
223 return err;
224}

266/**
267 * fb_get_vblank - get vertical blank time
268 * @hfreq: horizontal freq
269 *
270 * DESCRIPTION:
271 * vblank = right_margin + vsync_len + left_margin
272 *
273 * given: right_margin = 1 (V_FRONTPORCH)
274 * vsync_len = 3
275 * flyback = 550
276 *
277 * flyback * hfreq
278 * left_margin = --------------- - vsync_len
279 * 1000000
280 */
281static u32 fb_get_vblank(u32 hfreq)
282{
283 u32 vblank;
284
285 vblank = (hfreq * FLYBACK550)/1000;
286 vblank = (vblank + 500)/1000;
287 return (vblank + V_FRONTPORCH1);
288}

290/**
291 * fb_get_hblank_by_freq - get horizontal blank time given hfreq
292 * @hfreq: horizontal freq
293 * @xres: horizontal resolution in pixels
294 *
295 * DESCRIPTION:
296 *
297 * xres * duty_cycle
298 * hblank = ------------------
299 * 100 - duty_cycle
300 *
301 * duty cycle = percent of htotal assigned to inactive display
302 * duty cycle = C - (M/Hfreq)
303 *
304 * where: C = ((offset - scale factor) * blank_scale)
305 * -------------------------------------- + scale factor
306 * 256
307 * M = blank_scale * gradient
308 *
309 */
310static u32 fb_get_hblank_by_hfreq(u32 hfreq, u32 xres)
311{
312 u32 c_val, m_val, duty_cycle, hblank;
313
314 c_val = (((H_OFFSET40 - H_SCALEFACTOR20) * H_BLANKSCALE128)/256 +
315 H_SCALEFACTOR20) * 1000;
316 m_val = (H_BLANKSCALE128 * H_GRADIENT600)/256;
317 m_val = (m_val * 1000000)/hfreq;
318 duty_cycle = c_val - m_val;
319 hblank = (xres * duty_cycle)/(100000 - duty_cycle);
320 return (hblank);
321}

323/**
324 * fb_get_hfreq - estimate hsync
325 * @vfreq: vertical refresh rate
326 * @yres: vertical resolution
327 *
328 * DESCRIPTION:
329 *
330 * (yres + front_port) * vfreq * 1000000
331 * hfreq = -------------------------------------
332 * (1000000 - (vfreq * FLYBACK)
333 *
334 */
335
336static u32 fb_get_hfreq(u32 vfreq, u32 yres)
337{
338 u32 divisor, hfreq;
339
340 divisor = (1000000 - (vfreq * FLYBACK550))/1000;
341 hfreq = (yres + V_FRONTPORCH1) * vfreq * 1000;
342 return (hfreq/divisor);
343}

345static void fb_timings_vfreq(struct __fb_timings *timings)
346{
347 timings->hfreq = fb_get_hfreq(timings->vfreq, timings->vactive);
348 timings->vblank = fb_get_vblank(timings->hfreq);
349 timings->vtotal = timings->vactive + timings->vblank;
350 timings->hblank = fb_get_hblank_by_hfreq(timings->hfreq,
351 timings->hactive);
352 timings->htotal = timings->hactive + timings->hblank;
353 timings->dclk = timings->htotal * timings->hfreq;
354}

356/*
357 * fb_get_mode - calculates video mode using VESA GTF
358 * @flags: if: 0 - maximize vertical refresh rate
359 * 1 - vrefresh-driven calculation;
360 * 2 - hscan-driven calculation;
361 * 3 - pixelclock-driven calculation;
362 * @val: depending on @flags, ignored, vrefresh, hsync or pixelclock
363 * @var: pointer to fb_var_screeninfo
364 * @info: pointer to fb_info
365 *
366 * DESCRIPTION:
367 * Calculates video mode based on monitor specs using VESA GTF.
368 * The GTF is best for VESA GTF compliant monitors but is
369 * specifically formulated to work for older monitors as well.
370 *
371 * If @flag==0, the function will attempt to maximize the
372 * refresh rate. Otherwise, it will calculate timings based on
373 * the flag and accompanying value.
374 *
375 * If FB_IGNOREMON bit is set in @flags, monitor specs will be
376 * ignored and @var will be filled with the calculated timings.
377 *
378 * All calculations are based on the VESA GTF Spreadsheet
379 * available at VESA's public ftp (http://www.vesa.org).
380 *
381 * NOTES:
382 * The timings generated by the GTF will be different from VESA
383 * DMT. It might be a good idea to keep a table of standard
384 * VESA modes as well. The GTF may also not work for some displays,
385 * such as, and especially, analog TV.
386 *
387 * REQUIRES:
388 * A valid info->monspecs, otherwise 'safe numbers' will be used.
389 */
390static int fb_get_mode(int flags, u32 val, struct fb_videomode *var)
391{
392 struct __fb_timings *timings;
393 u32 interlace = 1, dscan = 1;
394 u32 hfmin, hfmax, vfmin, vfmax, dclkmin, dclkmax, err = 0;
395
396 timings = xzalloc(sizeof(struct __fb_timings));
397
398 /*
399 * If monspecs are invalid, use values that are enough
400 * for 640x480@60
401 */
402 hfmin = 29000; hfmax = 30000;
403 vfmin = 60; vfmax = 60;
404 dclkmin = 0; dclkmax = 25000000;
Value stored to 'dclkmax' is never read
405
406 timings->hactive = var->xres;
407 timings->vactive = var->yres;
408 if (var->vmode & FB_VMODE_INTERLACED1) {
409 timings->vactive /= 2;
410 interlace = 2;
411 }
412 if (var->vmode & FB_VMODE_DOUBLE2) {
413 timings->vactive *= 2;
414 dscan = 2;
415 }
416
417 /* vrefresh driven */
418 timings->vfreq = val;
419 fb_timings_vfreq(timings);
420
421 if (timings->dclk)
422 var->pixclock = KHZ2PICOS(timings->dclk / 1000)(1000000000UL/(timings->dclk / 1000));
423 var->hsync_len = (timings->htotal * 8) / 100;
424 var->right_margin = (timings->hblank / 2) - var->hsync_len;
425 var->left_margin = timings->hblank - var->right_margin -
426 var->hsync_len;
427 var->vsync_len = (3 * interlace) / dscan;
428 var->lower_margin = (1 * interlace) / dscan;
429 var->upper_margin = (timings->vblank * interlace) / dscan -
430 (var->vsync_len + var->lower_margin);
431
432 freebarebox_free(timings);
433 return err;
434}

631static int get_std_timing(unsigned char *block, struct fb_videomode *mode,
632 int ver, int rev)
633{
634 int xres, yres = 0, refresh, ratio, i;
635
636 xres = (block[0] + 31) * 8;
637 if (xres <= 256)
638 return 0;
639
640 ratio = (block[1] & 0xc0) >> 6;
641 switch (ratio) {
642 case 0:
643 /* in EDID 1.3 the meaning of 0 changed to 16:10 (prior 1:1) */
644 if (ver < 1 || (ver == 1 && rev < 3))
645 yres = xres;
646 else
647 yres = (xres * 10) / 16;
648 break;
649 case 1:
650 yres = (xres * 3) / 4;
651 break;
652 case 2:
653 yres = (xres * 4) / 5;
654 break;
655 case 3:
656 yres = (xres * 9) / 16;
657 break;
658 }
659 refresh = (block[1] & 0x3f) + 60;
660
661 for (i = 0; i < VESA_MODEDB_SIZE(sizeof(vesa_modes) / sizeof((vesa_modes)[0]) + (sizeof(struct
{ int:(-!!(__builtin_types_compatible_p(typeof((vesa_modes))
, typeof(&(vesa_modes)[0])))); }))); i++) {
662 if (vesa_modes[i].xres == xres &&
663 vesa_modes[i].yres == yres &&
664 vesa_modes[i].refresh == refresh) {
665 add_vesa_mode(mode, i);
666 return 1;
667 }
668 }
669
670 calc_mode_timings(xres, yres, refresh, mode);
671
672 return 1;
673}

728/**
729 * edid_to_display_timings - create video mode database
730 * @edid: EDID data
731 * @dbsize: database size
732 *
733 * RETURNS: struct fb_videomode, @dbsize contains length of database
734 *
735 * DESCRIPTION:
736 * This function builds a mode database using the contents of the EDID
737 * data
738 */
739int edid_to_display_timings(struct display_timings *timings, unsigned char *edid)
740{
741 struct fb_videomode *mode;
742 unsigned char *block;
743 int num = 0, i, first = 1;
744 int ver, rev, ret;
745
746 ver = edid[EDID_STRUCT_VERSION0x12];
747 rev = edid[EDID_STRUCT_REVISION0x13];
748
749 mode = xzalloc(50 * sizeof(struct fb_videomode));
750
751 if (!edid_checksum(edid) ||
752 !edid_check_header(edid)) {
753 ret = -EINVAL22;
754 goto out;
755 }
756
757 pr_debug(" Detailed Timings\n")({ (7) <= 6 ? pr_print((7), ("EDID: " " Detailed Timings\n"
)) : 0; });
758 block = edid + DETAILED_TIMING_DESCRIPTIONS_START0x36;
759 for (i = 0; i < 4; i++, block+= DETAILED_TIMING_DESCRIPTION_SIZE18) {
760 if (!(block[0] == 0x00 && block[1] == 0x00)) {
761 get_detailed_timing(block, &mode[num]);
762 if (first) {
763 first = 0;
764 }
765 num++;
766 }
767 }
768
769 pr_debug(" Supported VESA Modes\n")({ (7) <= 6 ? pr_print((7), ("EDID: " " Supported VESA Modes\n"
)) : 0; });
770 block = edid + ESTABLISHED_TIMING_10x23;
771 num += get_est_timing(block, &mode[num]);
772
773 pr_debug(" Standard Timings\n")({ (7) <= 6 ? pr_print((7), ("EDID: " " Standard Timings\n"
)) : 0; });
774 block = edid + STD_TIMING_DESCRIPTIONS_START0x26;
775 for (i = 0; i < STD_TIMING8; i++, block += STD_TIMING_DESCRIPTION_SIZE2)
776 num += get_std_timing(block, &mode[num], ver, rev);
777
778 block = edid + DETAILED_TIMING_DESCRIPTIONS_START0x36;
779 for (i = 0; i < 4; i++, block+= DETAILED_TIMING_DESCRIPTION_SIZE18) {

800#define DDC_SEGMENT_ADDR0x30 0x30

802/**
803 * Get EDID information via I2C.
804 *
805 * \param adapter : i2c device adaptor
806 * \param buf : EDID data buffer to be filled
807 * \param len : EDID data buffer length
808 * \return 0 on success or -1 on failure.
809 *
810 * Try to fetch EDID information by calling i2c driver function.
811 */
812static int
813edid_do_read_i2c(struct i2c_adapter *adapter, unsigned char *buf,
814 int block, int len)
815{
816 unsigned char start = block * EDID_LENGTH0x80;
817 unsigned char segment = block >> 1;
818 unsigned char xfers = segment ? 3 : 2;
819 int ret, retries = 5;
820
821 /* The core i2c driver will automatically retry the transfer if the
822 * adapter reports EAGAIN. However, we find that bit-banging transfers
823 * are susceptible to errors under a heavily loaded machine and
824 * generate spurious NAKs and timeouts. Retrying the transfer
825 * of the individual block a few times seems to overcome this.
826 */
827 do {
828 struct i2c_msg msgs[] = {
829 {
830 .addr = DDC_SEGMENT_ADDR0x30,
831 .flags = 0,
832 .len = 1,
833 .buf = &segment,
834 }, {
835 .addr = DDC_ADDR0x50,
836 .flags = 0,
837 .len = 1,
838 .buf = &start,
839 }, {
840 .addr = DDC_ADDR0x50,
841 .flags = I2C_M_RD0x0001,
842 .len = len,
843 .buf = buf,
844 }
845 };
846
847 /*
848 * Avoid sending the segment addr to not upset non-compliant ddc
849 * monitors.
850 */
851 ret = i2c_transfer(adapter, &msgs[3 - xfers], xfers);
852 } while (ret != xfers && --retries);
853
854 return ret == xfers ? 0 : -1;
855}
856
857void *edid_read_i2c(struct i2c_adapter *adapter)
858{
859 u8 *block;
860
861 if (!IS_ENABLED(CONFIG_I2C)1)
862 return NULL((void *)0);
863
864 block = xmalloc(EDID_LENGTH0x80);
865
866 if (edid_do_read_i2c(adapter, block, 0, EDID_LENGTH0x80))
867 goto out;
868
869 return block;
870out:
871 freebarebox_free(block);
872
873 return NULL((void *)0);
874}
