444 dm9k_iow(priv, DM9K_FCR0x0A, 0x00);
445 /* Special Mode */
446 dm9k_iow(priv, DM9K_SMCR0x2F, 0x00);
447 /* clear TX status */
448 dm9k_iow(priv, DM9K_NSR0x01, NSR_WAKEST(1 << 5) | NSR_TX2END(1 << 3) | NSR_TX1END(1 << 2));
449 /* Clear interrupt status */
450 dm9k_iow(priv, DM9K_IMR0xFF, IMR_PAR(1 << 7));
451 dm9k_iow(priv, DM9K_ISR0xFE, ISR_CLEAR_MASK((1 << 0) | (1 << 1) | (1 << 2) | (1 <<
3) | (1 << 4) | (1 << 5)));
452
453 /* Activate DM9000 */
454 dm9k_iow(priv, DM9K_GPCR0x1e, 0x01); /* Let GPIO0 output */
455 dm9k_iow(priv, DM9K_GPR0x1f, 0x00); /* Enable PHY */
456 /* RX enable */
457 dm9k_iow(priv, DM9K_RCR0x05, RCR_DIS_LONG(1 << 5) | RCR_DIS_CRC(1 << 4) | RCR_RXEN(1 << 0));
458 /* Enable TX/RX interrupt mask */
459 dm9k_iow(priv, DM9K_IMR0xFF, IMR_PAR(1 << 7) | IMR_PRM(1 << 0) | IMR_PTM(1 << 1));
460}

462static void dm9k_reset(struct dm9k *priv)
463{
464 struct device_d *dev = priv->miibus.parent;
465
466 dev_dbg(dev, "%s\n", __func__)({ (7) <= 6 ? dev_printf((7), ((dev)), ("%s\n"), __func__)
: 0; });
467
468 /* Reset DM9000, see DM9000 Application Notes V1.22 Jun 11, 2004 page 29
469 * The essential point is that we have to do a double reset, and the
470 * instruction is to set LBK into MAC internal loopback mode.
471 */
472
473 /* Make all GPIO pins outputs */
474 dm9k_iow(priv, DM9K_GPCR0x1e, 0x0F);
475 /* Power internal PHY by writing 0 to GPIO0 pin */
476 dm9k_iow(priv, DM9K_GPR0x1f, 0);
477
478 dm9k_iow(priv, DM9K_NCR0x00, NCR_RST(1 << 0) | NCR_MAC_LBK(1 << 1));
479 udelay(100); /* Application note says at least 20 us */
480 if (dm9k_ior(priv, DM9K_NCR0x00) & NCR_RST(1 << 0))
481 dev_err(dev, "dm9000 did not respond to first reset\n")({ (3) <= 6 ? dev_printf((3), ((dev)), ("dm9000 did not respond to first reset\n"
)) : 0; });
482
483 dm9k_iow(priv, DM9K_NCR0x00, 0);
484 dm9k_iow(priv, DM9K_NCR0x00, NCR_RST(1 << 0) | NCR_MAC_LBK(1 << 1));
485 udelay(100);
486
487 if (dm9k_ior(priv, DM9K_NCR0x00) & NCR_RST(1 << 0))
488 dev_err(dev, "dm9000 did not respond to second reset\n")({ (3) <= 6 ? dev_printf((3), ((dev)), ("dm9000 did not respond to second reset\n"
)) : 0; });
489}

505static int dm9k_wait_for_trans_end(struct dm9k *priv)
506{
507 struct device_d *dev = priv->miibus.parent;
508 static const uint64_t toffs = 1 * SECOND((uint64_t)(1000 * 1000 * 1000));
509 uint8_t status;
510 uint64_t start = get_time_ns();
511
512 do {
513 status = dm9k_ior(priv, DM9K_NSR0x01);
514 if (status & (NSR_TX1END(1 << 2) | NSR_TX2END(1 << 3))) {
515 dev_dbg(dev, "transmit done\n")({ (7) <= 6 ? dev_printf((7), ((dev)), ("transmit done\n")
) : 0; });
516 return 0;
517 }
518 status = dm9k_ior(priv, DM9K_ISR0xFE);
519 if (status & IMR_PTM(1 << 1)) {
520 /* Clear Tx bit in ISR */
521 dm9k_iow(priv, DM9K_ISR0xFE, IMR_PTM(1 << 1));
522 dev_dbg(dev, "transmit done\n")({ (7) <= 6 ? dev_printf((7), ((dev)), ("transmit done\n")
) : 0; });
523 return 0;
524 }
525 } while (!is_timeout(start, toffs));
526
527 return -ETIMEDOUT110;
528}

530static int dm9k_eth_send(struct eth_device *edev, void *packet, int length)
531{
532 struct dm9k *priv = (struct dm9k *)edev->priv;
533 struct device_d *dev = priv->miibus.parent;
534
535 dev_dbg(dev, "%s: %d bytes\n", __func__, length)({ (7) <= 6 ? dev_printf((7), ((dev)), ("%s: %d bytes\n"),
__func__, length) : 0; });
536
537 /* arm the Tx bit */
538 dm9k_iow(priv, DM9K_ISR0xFE, IMR_PTM(1 << 1));
539
540 /* Prepare for TX-data */
541 writeb(DM9K_MWCMD, priv->iobase)((void)0, *(volatile unsigned char *)(priv->iobase) = (0xF8
));
542
543 /* Move the packet into the DM9k's TX RAM */
544 dm9k_wd(priv->buswidth, priv->iodata, packet, length);
545
546 /* Set TX length of the packet */
547 dm9k_write_length(priv, length);
548
549 /* Issue TX polling command */
550 dm9k_iow(priv, DM9K_TCR0x02, TCR_TXREQ(1 << 0)); /* Cleared after TX complete */
551
552 /* wait for end of transmission */
553 return dm9k_wait_for_trans_end(priv);
554}

569static int dm9k_validate_entry(struct dm9k *priv)
570{
571 struct device_d *dev = priv->miibus.parent;
572
573 uint8_t p_stat;
574 /*
575 * setup read pointer to current packet
576 * but without address increment
577 */
578 dm9k_ior(priv, DM9K_MRCMDX0xF0);
579
580 /* read the entry's status according to the app note */
581 p_stat = readb(priv->iodata)((void)0, *(volatile unsigned char *)(priv->iodata)) & 0x03;
582 dev_dbg(dev, "%s packet status %02X\n", __func__, p_stat)({ (7) <= 6 ? dev_printf((7), ((dev)), ("%s packet status %02X\n"
), __func__, p_stat) : 0; });
583
584 switch (p_stat) {
585 case DM9K_PKT_NONE0x00: /* there is no packet (or the last in the chain) */
586 return 0;
587
588 case DM9K_PKT_ERR0x02: /* chip in invalid state. Needs a software reset */
589 dev_dbg(dev, "Confused chip.\n")({ (7) <= 6 ? dev_printf((7), ((dev)), ("Confused chip.\n"
)) : 0; });
590 dm9k_iow(priv, DM9K_RCR0x05, 0x00); /* Stop Device */
591 dm9k_iow(priv, DM9K_ISR0xFE, 0x80); /* Stop INT request */
592 dm9k_reset(priv);
593 dm9k_enable(priv);
594 return 0;
595 }
596
597 return 1; /* entry is valid */
598}

600static int dm9k_eth_rx(struct eth_device *edev)
601{
602 struct dm9k *priv = (struct dm9k *)edev->priv;
603 struct device_d *dev = edev->parent;
604 unsigned rx_stat = 0, rx_len = 0;
605 bool p_valid;
606
607 if (dm9k_check_for_rx_packet(priv) == 0)
608 return 0; /* no data present */
609
610 do {
611 if (!dm9k_validate_entry(priv)) {
612 dm9k_iow(priv, DM9K_ISR0xFE, ISR_PR(1 << 0)); /* clear PR status latched in bit 0 */
613 return 0;
614 }
615
616 /* assume this packet is valid */
617 p_valid = true;
618
619 /* read with automatic address increment now */
620 writeb(DM9K_MRCMD, priv->iobase)((void)0, *(volatile unsigned char *)(priv->iobase) = (0xF2
));
621 rx_len = dm9k_read_packet_status(priv->buswidth, priv->iodata, &rx_stat);
622 if (rx_len < 0x40) {
623 dev_dbg(dev, "Packet too short (%u bytes)\n", rx_len)({ (7) <= 6 ? dev_printf((7), ((dev)), ("Packet too short (%u bytes)\n"
), rx_len) : 0; });
624 p_valid = false;
625 }
626
627 /* validate packet */
628 if (rx_stat & RSR_ERR_MASK((1 << 0) | (1 << 1) | (1 << 2) | (1 <<
3) | (1 << 4) | (1 << 5) | (1 << 7))) {
629 if (rx_stat & RSR_FOE(1 << 0))
630 dev_warn(dev, "rx fifo overflow error\n")({ (4) <= 6 ? dev_printf((4), ((dev)), ("rx fifo overflow error\n"
)) : 0; });
631 if (rx_stat & RSR_CE(1 << 1))
632 dev_warn(dev, "rx crc error\n")({ (4) <= 6 ? dev_printf((4), ((dev)), ("rx crc error\n"))
: 0; });
633 if (rx_stat & RSR_AE(1 << 2))
634 dev_warn(dev, "rx Alignment Error error\n")({ (4) <= 6 ? dev_printf((4), ((dev)), ("rx Alignment Error error\n"
)) : 0; });
635 if (rx_stat & RSR_PLE(1 << 3))
636 dev_warn(dev, "rx Physical Layer Error error\n")({ (4) <= 6 ? dev_printf((4), ((dev)), ("rx Physical Layer Error error\n"
)) : 0; });
637 if (rx_stat & RSR_RWTO(1 << 4))
638 dev_warn(dev, "rx Receive Watchdog Time Out error\n")({ (4) <= 6 ? dev_printf((4), ((dev)), ("rx Receive Watchdog Time Out error\n"
)) : 0; });
639 if (rx_stat & RSR_LCS(1 << 5))
640 dev_warn(dev, "rx Late Collision Seen error\n")({ (4) <= 6 ? dev_printf((4), ((dev)), ("rx Late Collision Seen error\n"
)) : 0; });
641 if (rx_stat & RSR_RF(1 << 7))
642 dev_warn(dev, "rx length error (runt frame)\n")({ (4) <= 6 ? dev_printf((4), ((dev)), ("rx length error (runt frame)\n"
)) : 0; });
643 p_valid = false;
644 }
645
646 if (rx_len > DM9K_PKT_MAX1536) {
647 dev_warn(dev, "rx length too big\n")({ (4) <= 6 ? dev_printf((4), ((dev)), ("rx length too big\n"
)) : 0; });
648 /* discard packet */
649 dm9k_dump(priv->buswidth, priv->iodata, rx_len);
650 dm9k_reset(priv);
651 dm9k_enable(priv);
652 return 0;
653 }
654
655 if (p_valid == true) {
656 dev_dbg(dev, "Receiving packet\n")({ (7) <= 6 ? dev_printf((7), ((dev)), ("Receiving packet\n"
)) : 0; });
657 dm9k_rd(priv->buswidth, priv->iodata, priv->pckt, rx_len);
658 dev_dbg(dev, "passing %u bytes packet to upper layer\n", rx_len)({ (7) <= 6 ? dev_printf((7), ((dev)), ("passing %u bytes packet to upper layer\n"
), rx_len) : 0; });
659 net_receive(edev, priv->pckt, rx_len);
660 return 0;
661 } else {
662 dev_dbg(dev, "Discarding packet\n")({ (7) <= 6 ? dev_printf((7), ((dev)), ("Discarding packet\n"
)) : 0; });
663 dm9k_dump(priv->buswidth, priv->iodata, rx_len); /* discard packet */
664 }
665 } while (1);
666
667 return 0;
668}

723static int dm9000_parse_dt(struct device_d *dev, struct dm9k *priv)
724{
725 struct device_node *np = dev->device_node;
726 uint32_t prop;
727
728 if (!IS_ENABLED(CONFIG_OFDEVICE)1 || !np)
729 return -ENODEV19;
730
731 if (of_property_read_bool(np, "davicom,no-eeprom")) {
732 priv->srom = 0;
733 } else {
734 priv->srom = 1;
735 }
736
737 if (of_property_read_u32(np, "reg-io-width", &prop)) {
738 /* Use 8-bit registers by default */
739 prop = 1;
740 }
741
742 switch (prop) {
743 case 1:
744 priv->buswidth = IORESOURCE_MEM_8BIT(0<<3);
745 break;
746 case 2:
747 priv->buswidth = IORESOURCE_MEM_16BIT(1<<3);
748 break;
749 case 4:
750 priv->buswidth = IORESOURCE_MEM_32BIT(3<<3);
751 break;
752 default:
753 dev_err(dev, "Wrong io resource size\n")({ (3) <= 6 ? dev_printf((3), ((dev)), ("Wrong io resource size\n"
)) : 0; });
754 return -EINVAL22;
755 }
756
757 return 0;
758}

771static int dm9k_probe(struct device_d *dev)
772{
773 struct resource *iores;
774 unsigned io_mode;
775 struct eth_device *edev;
776 struct dm9k *priv;
777 int ret;
778
779 if (dev->num_resources < 2) {
780 dev_err(dev, "Need 2 resources base and data")({ (3) <= 6 ? dev_printf((3), ((dev)), ("Need 2 resources base and data"
)) : 0; });
781 return -ENODEV19;
782 }
783
784 edev = xzalloc(sizeof(struct eth_device) + sizeof(struct dm9k));
785 edev->priv = (struct dm9k *)(edev + 1);
786 priv = edev->priv;
787
788 if (dev->platform_data) {
789 ret = dm9000_parse_pdata(dev, priv);
790 } else {
791 ret = dm9000_parse_dt(dev, priv);
792 }
793
794 if (ret)
795 goto err;
796
797 iores = dev_request_mem_resource(dev, 1);
798 if (IS_ERR(iores)) {
799 ret = PTR_ERR(iores);
800 goto err;
801 }
802 priv->iodata = IOMEM(iores->start)((void *)(iores->start));
803
804 iores = dev_request_mem_resource(dev, 0);
805 if (IS_ERR(iores)) {
806 ret = PTR_ERR(iores);
807 goto err;
808 }
809 priv->iobase = IOMEM(iores->start)((void *)(iores->start));
810
811 edev->init = dm9k_init_dev;
812 edev->openbarebox_open = dm9k_eth_open;
813 edev->send = dm9k_eth_send;
814 edev->recv = dm9k_eth_rx;
815 edev->halt = dm9k_eth_halt;
816 edev->set_ethaddr = dm9k_set_ethaddr;
817 edev->get_ethaddr = dm9k_get_ethaddr;
818 edev->parent = dev;
819
820 priv->miibus.readbarebox_read = dm9k_phy_read;
821 priv->miibus.writebarebox_write = dm9k_phy_write;
822 priv->miibus.priv = priv;
823 priv->miibus.parent = dev;
824
825 /* RESET device */
826 dm9k_reset(priv);
827 if (dm9k_check_id(priv)) {
828 ret = -ENODEV19;
829 goto err;
830 }
831
832 io_mode = dm9k_ior(priv, DM9K_ISR0xFE) >> 6;
833 switch (io_mode) {
834 case 0:
835 dev_dbg(dev, "16 bit data bus\n")({ (7) <= 6 ? dev_printf((7), ((dev)), ("16 bit data bus\n"
)) : 0; });

)) : 0; })
838 "Wrong databus width defined at compile time\n")({ (3) <= 6 ? dev_printf((3), ((dev)), ("Wrong databus width defined at compile time\n"
)) : 0; });
839 break;
840 case 1:
841 dev_dbg(dev, "32 bit data bus\n")({ (7) <= 6 ? dev_printf((7), ((dev)), ("32 bit data bus\n"
)) : 0; });
842 if (priv->buswidth != IORESOURCE_MEM_32BIT(3<<3))
843 dev_err(dev,({ (3) <= 6 ? dev_printf((3), ((dev)), ("Wrong databus width defined at compile time\n"
)) : 0; })
844 "Wrong databus width defined at compile time\n")({ (3) <= 6 ? dev_printf((3), ((dev)), ("Wrong databus width defined at compile time\n"
)) : 0; });
845 break;
846 case 2:
847 dev_dbg(dev, "8 bit data bus\n")({ (7) <= 6 ? dev_printf((7), ((dev)), ("8 bit data bus\n"
)) : 0; });
848 if (priv->buswidth != IORESOURCE_MEM_8BIT(0<<3))
849 dev_err(dev,({ (3) <= 6 ? dev_printf((3), ((dev)), ("Wrong databus width defined at compile time\n"
)) : 0; })
850 "Wrong databus width defined at compile time\n")({ (3) <= 6 ? dev_printf((3), ((dev)), ("Wrong databus width defined at compile time\n"
)) : 0; });
851 break;
852 default:
853 dev_warn(dev, "Unknown data width\n")({ (4) <= 6 ? dev_printf((4), ((dev)), ("Unknown data width\n"
)) : 0; });

863err:

865
866 return ret;
867}
