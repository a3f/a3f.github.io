1168/**
1169 * mtd_ecclayout_ecc - Default ooblayout_ecc iterator implementation
1170 * @mtd: MTD device structure
1171 * @section: ECC section. Depending on the layout you may have all the ECC
1172 * bytes stored in a single contiguous section, or one section
1173 * per ECC chunk (and sometime several sections for a single ECC
1174 * ECC chunk)
1175 * @oobecc: OOB region struct filled with the appropriate ECC position
1176 * information
1177 *
1178 * This function is just a wrapper around the mtd->ecclayout field and is
1179 * here to ease the transition to the mtd_ooblayout_ops approach.
1180 * All it does is convert the layout->eccpos information into proper oob
1181 * region definitions.
1182 *
1183 * Returns zero on success, a negative error code otherwise.
1184 */
1185static int mtd_ecclayout_ecc(struct mtd_info *mtd, int section,
1186 struct mtd_oob_region *oobecc)
1187{
1188 int eccbyte = 0, cursection = 0, length = 0, eccpos = 0;
1189
1190 if (!mtd->ecclayout)
1191 return -ENOTSUPP524;
1192
1193 /*
1194 * This logic allows us to reuse the ->ecclayout information and
1195 * expose them as ECC regions (as done for the OOB free regions).
1196 *
1197 * TODO: this should be dropped as soon as we get rid of the
1198 * ->ecclayout field.
1199 */
1200 for (eccbyte = 0; eccbyte < mtd->ecclayout->eccbytes; eccbyte++) {
1201 eccpos = mtd->ecclayout->eccpos[eccbyte];
1202
1203 if (eccbyte < mtd->ecclayout->eccbytes - 1) {
1204 int neccpos = mtd->ecclayout->eccpos[eccbyte + 1];
1205
1206 if (eccpos + 1 == neccpos) {
1207 length++;
1208 continue;
1209 }
1210 }
1211
1212 if (section == cursection)
1213 break;
1214
1215 length = 0;
1216 cursection++;
1217 }
1218
1219 if (cursection != section || eccbyte >= mtd->ecclayout->eccbytes)
1220 return -ERANGE34;
1221
1222 oobecc->length = length + 1;
1223 oobecc->offset = eccpos - length;
1224
1225 return 0;
1226}
