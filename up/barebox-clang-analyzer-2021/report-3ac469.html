710 * @hsotg: Programming view of DWC_otg controller
711 */

713{
714 u32 gsnpsid;
715 u32 ghwcfg4;
716
717 /* Check if core configuration includes the IDDIG filter. */
718 ghwcfg4 = dwc2_readl(dwc2, GHWCFG4(0x0050));
719 if (!(ghwcfg4 & GHWCFG4_IDDIG_FILT_EN(1UL << (20))))
720 return false;

723 * Check if the IDDIG debounce filter is bypassed. Available
724 * in core version >= 3.10a.
725 */
726 gsnpsid = dwc2_readl(dwc2, GSNPSID(0x0040));
727 if (gsnpsid >= DWC2_CORE_REV_3_10a0x4f54310a) {
728 u32 gotgctl = dwc2_readl(dwc2, GOTGCTL(0x000));

731 return false;

737/*
738 * Do core a soft reset of the core. Be careful with this because it
739 * resets all the internal state machines of the core.
740 */
741int dwc2_core_reset(struct dwc2 *dwc2)
742{
743 bool wait_for_host_mode = false;
744 uint32_t greset;
745 int ret;
746
747 dwc2_dbg(dwc2, "%s(%p)\n", __func__, dwc2)({ (7) <= 6 ? dev_printf((7), (((dwc2)->dev)), ("%s(%p)\n"
), __func__, dwc2) : 0; });
748
749 /* Wait for AHB master IDLE state. */
750 ret = dwc2_wait_bit_set(dwc2, GRSTCTL(0x010), GRSTCTL_AHBIDLE(1UL << (31)), 10000);
751 if (ret) {
752 dwc2_warn(dwc2, "%s: Timeout! Waiting for AHB master IDLE state\n",({ (3) <= 6 ? dev_printf((3), (((dwc2)->dev)), ("%s: Timeout! Waiting for AHB master IDLE state\n"
), __func__) : 0; })
753 __func__)({ (3) <= 6 ? dev_printf((3), (((dwc2)->dev)), ("%s: Timeout! Waiting for AHB master IDLE state\n"
), __func__) : 0; });
754 return ret;
755 }
756
757 /*
758 * If the current mode is host, either due to the force mode
759 * bit being set (which persists after core reset) or the
760 * connector id pin, a core soft reset will temporarily reset
761 * the mode to device. A delay from the IDDIG debounce filter
762 * will occur before going back to host mode.
763 *
764 * Determine whether we will go back into host mode after a
765 * reset and account for this delay after the reset.
766 */
767 if (dwc2_iddig_filter_enabled(dwc2)) {
768 u32 gotgctl = dwc2_readl(dwc2, GOTGCTL(0x000));
769 u32 gusbcfg = dwc2_readl(dwc2, GUSBCFG(0x00C));
770
771 if (!(gotgctl & GOTGCTL_CONID_B(1UL << (16))) ||
772 (gusbcfg & GUSBCFG_FORCEHOSTMODE(1UL << (29)))) {
773 wait_for_host_mode = true;
774 }
775 }
776
777 /* Core Soft Reset */
778 greset = dwc2_readl(dwc2, GRSTCTL(0x010));
779 greset |= GRSTCTL_CSFTRST(1UL << (0));
780 dwc2_writel(dwc2, greset, GRSTCTL(0x010));
781
782 ret = dwc2_wait_bit_clear(dwc2, GRSTCTL(0x010), GRSTCTL_CSFTRST(1UL << (0)), 10000);
783 if (ret) {
784 dwc2_warn(dwc2, "%s: Timeout! Waiting for Core Soft Reset\n",({ (3) <= 6 ? dev_printf((3), (((dwc2)->dev)), ("%s: Timeout! Waiting for Core Soft Reset\n"
), __func__) : 0; })
785 __func__)({ (3) <= 6 ? dev_printf((3), (((dwc2)->dev)), ("%s: Timeout! Waiting for Core Soft Reset\n"
), __func__) : 0; });
786 return ret;
787 }
788
789 if (wait_for_host_mode)
790 dwc2_wait_for_mode(dwc2, wait_for_host_mode);
791
792 return 0;
793}
