72/**
73 * struct xyz_ctxt - context of a x/y modem (g) transfer
74 *
75 * @cdev: console device to support *MODEM transfer
76 * @fifo: fifo to buffer input from serial line
77 * This is necessary for low hardware FIFOs buffers as UARTs.
78 * @mode: protocol (XMODEM, YMODEM or YMODEM/G)
79 * @crc_mode: CRC_NONE, CRC_ADD8 or CRC_CRC16
80 * @state: protocol state (as in "state machine")
81 * @buf: buffer to store the last tranfered buffer chunk
82 * @filename : filename transmitted by sender (YMODEM* only)
83 * @fd : file descriptor of the current stored file
84 * @file_len: length declared by sender (YMODEM* only)
85 * @nb_received: number of data bytes received since session open
86 * (this doesn't count resends)
87 * @total_SOH: number of SOH frames received (128 bytes chunks)
88 * @total_STX: number of STX frames received (1024 bytes chunks)
89 * @total_CAN: nubmer of CAN frames received (cancel frames)
90 */
91struct xyz_ctxt {
92 struct console_device *cdev;
93 struct kfifo *fifo;
94 int mode;
95 int crc_mode;
96 enum proto_state state;
97 char filename[1024];
98 int fd;
99 int file_len;
100 int nb_received;
101 int next_blk;
102 int total_SOH, total_STX, total_CAN, total_retries;
103};

201static int check_crc(unsigned char *buf, int len, int crc, int crc_mode)
202{
203 unsigned char crc8 = 0;
204 uint16_t crc16;
205 int i;
206
207 switch (crc_mode) {
208 case CRC_ADD81:
209 for (i = 0; i < len; i++)
210 crc8 += buf[i];
211 return crc8 == crc ? 0 : -EBADMSG74;
212 case CRC_CRC162:
213 crc16 = crc_itu_t(0, buf, len);
214 xy_dbg("crc16: received = %x, calculated=%x\n", crc, crc16);
215 return crc16 == crc ? 0 : -EBADMSG74;
216 case CRC_NONE0:
217 return 0;
218 default:
219 return -EBADMSG74;
220 }
221}

223/**
224 * xy_read_block - read a X-Modem or Y-Modem(G) block
225 * @proto: protocol control structure
226 * @blk: block read
227 * @timeout: maximal time to get data
228 *
229 * This is the pivotal function for block receptions. It attempts to receive one
230 * block, ie. one 128 bytes or one 1024 bytes block. The received data can also
231 * be an end of transmission, or a cancel.
232 *
233 * Returns :
234 * >0 : size of the received block
235 * 0 : last block, ie. end of transmission, ie. EOT
236 * -EBADMSG : malformed message (ie. sequence bi-bytes are not
237 * complementary), or CRC check error
238 * -EILSEQ : block sequence number error wrt previously received block
239 * -ETIMEDOUT : block not received before timeout passed
240 * -ECONNABORTED : transfer aborted by sender, ie. CAN
241 */
242static ssize_t xy_read_block(struct xyz_ctxt *proto, struct xy_block *blk,
243 uint64_t timeout)
244{
245 ssize_t rc, data_len = 0;
246 unsigned char hdr, seqs[2], crcs[2];
247 int crc = 0;
248 bool hdr_found = 0;
249 uint64_t start = get_time_ns();
250
251 while (!hdr_found) {
252 rc = xy_gets(proto->cdev, proto->fifo, &hdr, 1, timeout);
253 xy_dbg("read 0x%x(%c) -> %d\n", hdr, hdr, rc);
254 if (rc < 0)
255 goto out;
256 if (is_timeout(start, timeout))
257 goto timeout;
258 switch (hdr) {
259 case SOH0x01:
260 data_len = 128;
261 hdr_found = 1;
262 proto->total_SOH++;
263 break;
264 case STX0x02:
265 data_len = 1024;
266 hdr_found = 1;
267 proto->total_STX++;
268 break;
269 case CAN0x18:
270 rc = -ECONNABORTED103;
271 if (proto->total_CAN++ > MAX_CAN_BEFORE_ABORT5)
272 goto out;
273 break;
274 case EOT0x04:
275 rc = 0;
276 blk->len = 0;
277 goto out;
278 default:
279 break;
280 }
281 }
282
283 blk->seq = 0;
284 rc = xy_gets(proto->cdev, proto->fifo, seqs, 2, timeout);
285 if (rc < 0)
286 goto out;
287 blk->seq = seqs[0];
288 if (255 - seqs[0] != seqs[1])
289 return -EBADMSG74;
290
291 rc = xy_gets(proto->cdev, proto->fifo, blk->buf, data_len, timeout);
292 if (rc < 0)
293 goto out;
294 blk->len = rc;
295
296 switch (proto->crc_mode) {
297 case CRC_ADD81:
298 rc = xy_gets(proto->cdev, proto->fifo, crcs, 1, timeout);
299 crc = crcs[0];
300 break;
301 case CRC_CRC162:
302 rc = xy_gets(proto->cdev, proto->fifo, crcs, 2, timeout);
303 crc = (crcs[0] << 8) + crcs[1];
304 break;
305 case CRC_NONE0:
306 rc = 0;
307 break;
308 }
309 if (rc < 0)
310 goto out;
311
312 rc = check_crc(blk->buf, data_len, crc, proto->crc_mode);
313 if (rc < 0)
314 goto out;
315 return data_len;
316timeout:
317 return -ETIMEDOUT110;
318out:
319 return rc;
320}

347static int xy_get_file_header(struct xyz_ctxt *proto)
348{
349 struct xy_block blk;
350 int tries, rc = 0;
351
352 memset(&blk, 0, sizeof(blk));
353 proto->state = PROTO_STATE_GET_FILENAME;
354 proto->crc_mode = CRC_CRC162;
355 for (tries = 0; tries < MAX_RETRIES10; tries++) {
356 xy_putc(proto->cdev,
357 invite_filename_hdr[proto->mode][proto->crc_mode]);
358 rc = xy_read_block(proto, &blk, 3 * SECOND((uint64_t)(1000 * 1000 * 1000)));
359 xy_dbg("read block returned %d\n", rc);
360 switch (rc) {
361 case -ECONNABORTED103:
362 goto fail;
363 case -ETIMEDOUT110:
364 case -EBADMSG74:
365 if (proto->mode != PROTO_YMODEM_G2)
366 xy_flush(proto->cdev, proto->fifo);
367 break;
368 case -EALREADY114:
369 default:
370 proto->next_blk = 1;
371 xy_block_ack(proto);
372 proto->state = PROTO_STATE_NEGOCIATE_CRC;
373 rc = parse_first_block(proto, &blk);
374 return rc;
375 }
376
377 if (rc < 0 && tries++ >= MAX_RETRIES_WITH_CRC5)
378 proto->crc_mode = CRC_ADD81;
379 }
380 rc = -ETIMEDOUT110;
381fail:
382 proto->total_retries += tries;
383 return rc;
