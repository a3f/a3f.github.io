396/**
397 * login() - Prompt for password
398 *
399 * This function only returns when the correct password has been entered or
400 * no password is necessary because either no password is configured or the
401 * correct password has been entered in a previous call to this function.
402 */
403void login(void)
404{
405 unsigned char passwd[PASSWD_MAX_LENGTH(128 + 1)];
406 int ret;
407
408 if (!is_passwd_default_enable() && !is_passwd_env_enable())
409 return;
410
411 if (logged_in)
412 return;
413
414 while (1) {
415 printfbarebox_printf("Password: ");
416
417 ret = password(passwd, PASSWD_MAX_LENGTH(128 + 1), LOGIN_MODE(1 << 1), login_timeout);
418 if (ret < 0)
419 run_command(login_fail_command);
420
421 if (ret < 0)
422 continue;
423
424 if (check_passwd(passwd, ret) != 1)
425 continue;
426
427 logged_in = 1;
428 return;
429 }
430}
