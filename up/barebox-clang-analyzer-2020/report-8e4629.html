383 host->mci.hw_dev = dev;

386 * rising edge ... meaning SPI modes 0 or 3. So either SPI mode
387 * should be legit. We'll use mode 0 since the steady state is 0,
388 * which is appropriate for hotplugging, unless the platform data
389 * specify mode 3 (if hardware is not compatible to mode 0).
390 */

398 spi->mode, spi->max_speed_hz / 1000,({ (7) <= 6 ? dev_printf((7), ((&spi->dev)), ("needs SPI mode %02x, %d KHz; %d\n"

399 status)({ (7) <= 6 ? dev_printf((7), ((&spi->dev)), ("needs SPI mode %02x, %d KHz; %d\n"

401 }
402
403 /* SPI doesn't need the lowspeed device identification thing for
404 * MMC or SD cards, since it never comes up in open drain mode.
405 * That's good; some SPI masters can't handle very low speeds!
406 *
407 * However, low speed SDIO cards need not handle over 400 KHz;
408 * that's the only reason not to use a few MHz for f_min (until
