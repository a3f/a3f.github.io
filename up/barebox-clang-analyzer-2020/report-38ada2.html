28/*
29 * I2C EEPROMs from most vendors are inexpensive and mostly interchangeable.
30 * Differences between different vendor product lines (like Atmel AT24C or
31 * MicroChip 24LC, etc) won't much matter for typical read/write access.
32 * There are also I2C RAM chips, likewise interchangeable. One example
33 * would be the PCF8570, which acts like a 24c02 EEPROM (256 bytes).
34 *
35 * However, misconfiguration can lose data. "Set 16-bit memory address"
36 * to a part with 8-bit addressing will overwrite data. Writing with too
37 * big a page size also loses data. And it's not safe to assume that the
38 * conventional addresses 0x50..0x57 only hold eeproms; a PCF8563 RTC
39 * uses 0x51, for just one example.
40 *
41 * So this driver uses "new style" I2C driver binding, expecting to be
42 * told what devices exist. That may be in arch/X/mach-Y/board-Z.c or
43 * similar kernel-resident tables; or, configuration data coming from
44 * a bootloader.
45 *
46 * Other than binding model, current differences from "eeprom" driver are
47 * that this one handles write access and isn't restricted to 24c02 devices.
48 * It also handles larger devices (32 kbit and up) with two-byte addresses,
49 * which won't work on pure SMBus systems.
50 */

73 * of I2C for potentially troublesome amounts of time. With a 100 kHz I2C

131static struct i2c_client *at24_translate_offset(struct at24_data *at24,
132 unsigned *offset)
133{
134 unsigned i;
135
136 if (at24->chip.flags & AT24_FLAG_ADDR160x80) {
137 i = *offset >> 16;
138 *offset &= 0xffff;
139 } else {
140 i = *offset >> 8;
141 *offset &= 0xff;
142 }
143
144 return at24->client[i];
145}

147static ssize_t at24_eeprom_read(struct at24_data *at24, char *buf,
148 unsigned offset, size_t count)
149{
150 struct i2c_msg msg[2];

154 uint64_t start, read_time;
155
156 memset(msg, 0, sizeof(msg));
157
158 /*
159 * REVISIT some multi-address chips don't rollover page reads to
160 * the next slave address, so we may need to truncate the count.
161 * Those chips might need another quirk flag.
162 *
163 * If the real hardware used four adjacent 24c02 chips and that
164 * were misconfigured as one 24c08, that would be a similar effect:
165 * one "eeprom" file not four, but larger reads would fail when
166 * they crossed certain pages.

176 if (count > io_limit)
177 count = io_limit;

179 i = 0;

181 msgbuf[i++] = offset >> 8;

195 * loop a few times until this one succeeds, waiting at least

202 if (status == 2)
203 status = count;

217static ssize_t at24_read(struct at24_data *at24,
218 char *buf, loff_t off, size_t count)
219{
220 ssize_t retval = 0;
221
222 if (unlikely(!count)__builtin_expect(!!(!count), 0))
223 return count;
224
225 /*
226 * Read data from chip, protecting against concurrent updates
227 * from this host, but not from other I2C masters.
228 */
229 while (count) {
230 ssize_t status;
231
232 status = at24_eeprom_read(at24, buf, off, count);

237 }
238 buf += status;
239 off += status;
240 count -= status;
241 retval += status;
242 }
243
244 return retval;
245}
